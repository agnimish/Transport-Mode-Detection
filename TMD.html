def helper_func(temp):
    temp = temp["timestamp"]
    temp = pd.DataFrame(data = temp).set_index("timestamp")
    temp["val"] = np.ones(len(temp))
    temp = temp.groupby(temp.index).count()
    temp = temp.resample('5s', origin='start').count()
    return len(temp[temp.val != 0])
